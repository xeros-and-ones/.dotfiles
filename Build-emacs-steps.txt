1-  https://savannah.gnu.org/git/?group=emacs

2-  cd emacs

3-  git worktree add ../my-custom-build (choose name you prefer)

4-  cd ../my-custom-build

5-  ./autogen.sh

6-  ./configure --with-cairo --with-x-toolkit=gtk3 --without-compress-install --with-harfbuzz --with-mailutils --with-native-compilation --with-json --without-gsettings --without-gconf

7-  make bootstrap -jN (N = number of cores you have in your cpu)

8- sudo make install.sh
